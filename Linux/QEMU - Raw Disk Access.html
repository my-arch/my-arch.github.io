QEMU/KVM

  1. sudo pacman -S spice-gtk ovmf qemu
  2. qemu-system-x86_64 -enable-kvm --bios /usr/share/ovmf/x64/OVMF_CODE.fd -m 8G -cpu host,hv_relaxed,hv_spinlocks=0x1fff,hv_vapic,hv_time -machine type=pc,accel=kvm -drive file=/dev/nvme0n1,format=raw,media=disk -smp $(nproc) -daemonize -vga qxl -spice port=5930,disable-ticketing -device virtio-serial -device virtserialport,chardev=spicechannel0,name=com.redhat.spice.0 -chardev spicevmc,id=spicechannel0,name=vdagent
  3. spicy -h 127.0.0.1 -p 5930 & disown spicy
  4. install spice driver in windows guest
